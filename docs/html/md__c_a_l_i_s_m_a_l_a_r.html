void show_menu_item(const char* item, int selected) {
lcd_clear();

lcd_string("> ");
} else {
lcd_string(" ");
}
lcd_string(item);
}

5. Hız Profili

void speed_profile(void) {
// Yavaşça hızlan
for(int i = 10; i <= 100; i += 10) {
step_turn(CW, i, 0.5);
}
// Yavaşça yavaşla
for(int i = 90; i >= 10; i -= 10) {
step_turn(CW, i, 0.5);
}
}

void display_counter(void) {
int counter = 0;
char buffer[16];
while(1) {
lcd_clear();
lcd_string("Sayac:");
lcd_set_cursor(1, 0);
sprintf(buffer, "%d", counter++);
lcd_string(buffer);
sleep_ms(1000);
}
}
