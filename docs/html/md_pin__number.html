Buzzer pin ayarları ve PWM nota çalma

#include "pico/stdlib.h"
#include "hardware/pwm.h"
// Define the buzzer pin
#define BUZZER_PIN 10
// Function to initialize PWM on the buzzer pin
void init_pwm() {
gpio_set_function(BUZZER_PIN, GPIO_FUNC_PWM);
uint slice_num = pwm_gpio_to_slice_num(BUZZER_PIN);
pwm_set_enabled(slice_num, true);
}
// Function to set the frequency of the PWM signal
void set_pwm_frequency(uint slice_num, float frequency) {
// Calculate the wrap value for the desired frequency
uint32_t clock = 125000000; // Default clock frequency for the Pico
uint32_t divider16 = (uint32_t)(clock / frequency / 4096);
if (((uint32_t)(clock / frequency)) %4096 != 0) {
divider16 += 1;
}
if (divider16 / 16 == 0) {
divider16 = 16;
}
pwm_set_clkdiv_int_frac(slice_num, divider16 / 16, divider16 & 0xF);
pwm_set_wrap(slice_num, 4095);
pwm_set_gpio_level(BUZZER_PIN, 2048); // 50% duty cycle
}
// Function to play a note for a specific duration
void play_note(float frequency, int duration_ms) {
uint slice_num = pwm_gpio_to_slice_num(BUZZER_PIN);
set_pwm_frequency(slice_num, frequency);
sleep_ms(duration_ms);
pwm_set_gpio_level(BUZZER_PIN, 0); // Turn off the buzzer
}
int main() {
stdio_init_all();
init_pwm();
// Example: Play a C note (261.63 Hz) for 500 ms
play_note(261.63, 500);
// Add more notes as needed
// play_note(frequency, duration);
return 0;
}
